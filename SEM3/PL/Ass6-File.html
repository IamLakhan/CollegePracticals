 83  /*Function to create data file initially */
 84  void create(int size)
 85  {
 86      int i;
 87      student s;
 88  
 89      fp = fopen(FILENAME, "w");
 90      for(i=0; i<size; i++)
 91      {
 92          printf("\nEnter Details For Student %d :",(i+1));
 93          printf("\nRoll No :");
 94          scanf("%d",&s.roll_no);
 95          printf("Name :");
 96          scanf("%s",s.name);
 97          printf("Marks :");
 98          scanf("%f", &s.marks);
 99  
100          fwrite((char *)&s,sizeof(student),1,fp);
101      }
102      fclose(fp);
103  }
104  
105  /*Function to display all data from file*/
106  void display(int size)
107  {
108      student s;
109      fp = fopen(FILENAME,"r");
110  
111      while(fread((char *)&s,sizeof(student),1,fp))
112      {
113          printf("\n%d\t\t%s\t\t%0.2f%%",s.roll_no,s.name,s.marks);
114      }
115      fclose(fp);
116  }

138  /*To search record using roll_no*/
139  int search(int key)
140  {
141      student s;
142      int r=-1;
143      fp = fopen(FILENAME,"r");
144      printf("\n==========================================");
145      printf("\nROLL_NO\t\tNAME\t\tMARKS");
146      printf("\n==========================================");
147      while(fread((char *)&s,sizeof(student),1,fp))
148      {
149          if(s.roll_no == key)
150          {
151              printf("\n%d\t\t%s\t\t%0.2f%%",s.roll_no,s.name,s.marks);
152              r=1;
153          }
154      }
155      if(r==-1)
156      {
157          printf("\nNO RECORD FOUND");
158      }
159      printf("\n==========================================");
160      fclose(fp);
161      return r;
162  }

199      student s;
